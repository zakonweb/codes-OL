Count=0, Time=0, FS=0, Total=0,
AVG=0
Highest=-1000, Lowest=1000
For Count = 1 to 500
   Input Time

   FS = 200/Time
   Output FS

   Total = Total + FS
   
   If FS > Highest Then 
     Highest = FS
   Endif

   If FS < Lowest Then 
     Lowest = FS
   Endif
Next
AVG = Total/500
Output Lowest, Highest, AVG
