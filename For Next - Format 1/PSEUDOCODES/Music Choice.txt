Count=0, r=0, s=0, p=0, j=0
c=0, Name="", PerR=0, PerS=0,
PerP=0, PerJ=0, PerC=0, Choice=0
For Count = 1 to 1500
 INPUT Choice, Name

 IF Choice =  1 THEN r=r+1
 IF Choice =  2 THEN s=s+1
 IF Choice =  3 THEN p=p+1
 IF Choice =  4 THEN j=j+1
 IF Choice =  5 THEN 
   c=c+1
   OUTPUT Name
 ENDIF 
Next Count
PerR = r/15
PerS = s/15
PerP = p/s15
PerJ = j/15
PerC = c/15
OUTPUT PerR, PerS, PerP, PerJ, PerC
