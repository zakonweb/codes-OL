Count=0, r=0, s=0, p=0, j=0
c=0, Name="", PerR=0, PerS=0,
PerP=0, PerJ=0, PerC=0, Choice=0
For Count = 1 to 1500
 INPUT Choice, Name

 CASE OF Choice
 1: r=r+1
 2: s=s+1
 3: p=p+1
 4: j=j+1
 5: 
   c=c+1
   OUTPUT Name
 ENDCASE
Next Count
PerR = r/15
PerS = s/15
PerP = p/s15
PerJ = j/15
PerC = c/15
OUTPUT PerR, PerS, PerP, PerJ, PerC
