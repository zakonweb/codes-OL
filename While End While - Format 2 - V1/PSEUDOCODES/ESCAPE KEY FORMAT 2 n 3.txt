INPUT SENSOR1, SENSOR2, KEY
WHILE (KEY <> ESCAPE)
  IF SENSOR1 > 45 THEN OUTPUT "Error"
  IF SENSOR2 < 0.19 THEN OUTPUT "Error"

  INPUT SENSOR1, SENSOR2, KEY
ENDWHILE
