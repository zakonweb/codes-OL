H=-1000

REPEAT
  INPUT ITEM
  IF ITEM = "bun" THEN B=B+0.5			
  IF ITEM = "coffee" THEN CO=CO+1.2		
  IF ITEM = "cake" THEN CA=CA+1.5		
  IF ITEM = "sandwich" THEN S=S+2.1		
  IF ITEM = "dessert" THEN D=D+4	
UNTIL ITEM = "end"

IF  B > HIGH THEN 
   HIG = B
   VAL = "bun"
ENDIF

IF CO > HIG   THEN 
   HIG = CO
   VAL = "coffee"
ENDIF

IF CA > HIG THEN 
   HIG = CA
   VAL = "cake"
ENDIF

IF S > HIGH THEN 
   HIG = S
   VAL = "sandwich"
ENDIF

IF D > HIG THEN 
   HIG = D
   VAL = "dessert"
ENDIF

TOTAL = B+CO+CA+S+D
OUTPUT TOTAL, VAL
