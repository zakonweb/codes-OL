day=0, Total=0, AvgD=0, reading=0,
Temp=0, Highest=-1000, Lowest=1000,
GTotal=0, AvgY=0

For day = 1 to 365
  Total=0, AvgD=0
  
  For reading = 1 to 10
    INPUT Temp
   
    Total = Total + Temp

    IF Temp > Highest THEN Highest = Temp
    IF Temp < Lowest THEN Lowest = Temp
  NEXT reading

  GTaotal = GTotal + Total

  AvgD = Total/10
  OUTPUT AvgD
Next day
AvgY = GTotal/3650
OUTPUT Highest, Lowest, AvgY
