Day = 0, Enq = 0, TEnq = 0, CustEnq = 0
A = 0, B = 0, Per = 0, HousePrice = 0

For Day = 1 to 7
  A=0

  INPUT Enq
  TEnq = Tenq + Enq

  For CustEnq = 1 to Enq
    INPUT HousePrice

    IF HousePrice < 100000 THEN A=A+1
    IF HousePrice > 500000 THEN B=B+1
  Next CustEnq

  OUTPUT A

Next Day
Per = (B/TEnq)*100
OUTPUT Per
